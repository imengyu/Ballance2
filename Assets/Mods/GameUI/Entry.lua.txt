GameManager = Ballance2.GameManager
GameLogger = Ballance2.GameLogger
GameEventNames = Ballance2.Managers.CoreBridge.GameEventNames
GameActionNames = Ballance2.Managers.CoreBridge.GameActionNames
ModManager = Ballance2.Managers.ModManager

local UIManager = GameManager.UIManager

TAG = "core.ui.Main"
gameModThis = nil

-- 模块入口函数
function ModEntry(thisGameMod)
	gameModThis = thisGameMod
	OnLoadAll()
	return true
end

-- 模块卸载前函数
function ModBeforeUnLoad(thisGameMod)
  gameModThis = thisGameMod
	return true
end

function OnLoadAll()

  gameModThis:RequireLuaFile("GameSettingsUI.lua.txt")

  LoadGameSettings()
  LoadGameSettingsUI()
  return true
  
end
